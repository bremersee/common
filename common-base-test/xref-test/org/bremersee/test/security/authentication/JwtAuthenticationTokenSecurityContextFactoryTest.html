19  /**
20   * The jwt authentication token security context factory test.
21   *
22   * @author Christian Bremer
23   */
24  @RunWith(SpringJUnit4ClassRunner.class)
25  @ContextConfiguration
26  public class JwtAuthenticationTokenSecurityContextFactoryTest {
27  
28    /**
29     * Create security context.
30     */
31    @WithJwtAuthenticationToken(
32        audience = "account",
33        issuer = "https://openid.dev.bremersee.org/auth/realms/omnia",
34        scope = {"email", "profile"},
35        jwtId = "080836fb-7e74-4a56-92ba-08aeaf9a3852",
36        subject = "1918e152-294b-4701-a2c8-b9090bb5aa06",
37        email = "plurabelle@example.net",
38        roles = {USER_ROLE_NAME, LOCAL_USER_ROLE_NAME})
39    @Test
40    public void createSecurityContext() {
41      Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
42      assertNotNull(authentication);
43      assertTrue(authentication instanceof KeycloakJwtAuthenticationToken);
44      KeycloakJwtAuthenticationToken authToken = (KeycloakJwtAuthenticationToken) authentication;
45      assertEquals("account", authToken.getToken().getClaimAsString("aud"));
46      assertEquals(
47          "https://openid.dev.bremersee.org/auth/realms/omnia",
48          authToken.getToken().getClaimAsString("iss"));
49      assertEquals(
50          "080836fb-7e74-4a56-92ba-08aeaf9a3852",
51          authToken.getToken().getClaimAsString("jti"));
52      assertEquals(
53          "1918e152-294b-4701-a2c8-b9090bb5aa06",
54          authToken.getToken().getClaimAsString("sub"));
55      assertEquals(
56          "Anna Livia Plurabelle",
57          authToken.getToken().getClaimAsString("name"));
58      assertEquals(
59          "anna",
60          authToken.getPreferredName());
61      assertEquals(
62          "Anna Livia",
63          authToken.getToken().getClaimAsString("given_name"));
64      assertEquals(
65          "Plurabelle",
66          authToken.getToken().getClaimAsString("family_name"));
67      assertEquals(
68          "plurabelle@example.net",
69          authToken.getToken().getClaimAsString("email"));
70      List<String> scopes = authToken.getToken().getClaimAsStringList("scope");
71      assertNotNull(scopes);
72      assertTrue(scopes.contains("email"));
73      assertTrue(scopes.contains("profile"));
74      assertTrue(authToken.getAuthorities().contains(new SimpleGrantedAuthority(USER_ROLE_NAME)));
75      assertTrue(
76          authToken.getAuthorities().contains(new SimpleGrantedAuthority(LOCAL_USER_ROLE_NAME)));
77    }
