17  package org.bremersee.security.authentication;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import net.minidev.json.JSONObject;
21  import net.minidev.json.JSONValue;
22  import org.bremersee.exception.AccessTokenRetrieverAuthenticationException;
23  import org.bremersee.web.ErrorDetectors;
24  import org.bremersee.web.reactive.function.client.AbstractWebClientErrorDecoder;
25  import org.springframework.http.HttpMethod;
26  import org.springframework.http.MediaType;
27  import org.springframework.security.core.AuthenticationException;
28  import org.springframework.web.reactive.function.BodyInserters;
29  import org.springframework.web.reactive.function.client.ClientResponse;
30  import org.springframework.web.reactive.function.client.WebClient;
31  import reactor.core.publisher.Mono;
32  
33  /**
34   * A reactive implementation of the {@link AccessTokenRetriever}.
35   *
36   * @author Christian Bremer
37   */
38  @Slf4j
39  public class WebClientAccessTokenRetriever
40      extends AbstractWebClientErrorDecoder<AuthenticationException>
41      implements AccessTokenRetriever<Mono<String>> {
42  
43    private final WebClient webClient;
44  
45    /**
46     * Instantiates a new access token retriever that uses spring's web client.
47     */
48    public WebClientAccessTokenRetriever() {
49      this.webClient = WebClient.builder().build();
50    }
51  
52    /**
53     * Instantiates a new access token retriever that uses spring's web client.
54     *
55     * @param webClient the web client
56     */
57    @SuppressWarnings("unused")
58    public WebClientAccessTokenRetriever(
59        final WebClient webClient) {
60      this.webClient = webClient;
61    }
62  
63    @Override
64    public Mono<String> retrieveAccessToken(final AccessTokenRetrieverProperties properties) {
65      if (log.isDebugEnabled()) {
66        log.debug("Retrieving access token with password flow, properties = {}", properties);
67      }
68      return webClient
69          .method(HttpMethod.POST)
70          .uri(properties.getTokenEndpoint())
71          .contentType(MediaType.APPLICATION_FORM_URLENCODED)
72          .headers(headers -> properties.getBasicAuthProperties()
73              .ifPresent(basicAuthProperties -> headers.setBasicAuth(
74                  basicAuthProperties.getUsername(),
75                  basicAuthProperties.getPassword())))
76          .body(BodyInserters.fromFormData(properties.createBody()))
77          .retrieve()
78          .onStatus(ErrorDetectors.DEFAULT, this)
79          .bodyToMono(String.class)
80          .map(response -> ((JSONObject) JSONValue.parse(response)).getAsString("access_token"));
81    }
82  
83    @Override
84    public AuthenticationException buildException(
85        final ClientResponse clientResponse, final String response) {
86      final AccessTokenRetrieverAuthenticationException exception
87          = new AccessTokenRetrieverAuthenticationException(clientResponse.statusCode(), response);
88      log.error("msg=[Retrieving access token with password flow failed.]", exception);
89      return exception;
90    }
91  
92  }
