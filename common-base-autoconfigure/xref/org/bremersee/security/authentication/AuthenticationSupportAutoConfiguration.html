17  package org.bremersee.security.authentication;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.springframework.beans.factory.ObjectProvider;
21  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
22  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
23  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
25  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication.Type;
26  import org.springframework.boot.context.event.ApplicationReadyEvent;
27  import org.springframework.boot.context.properties.EnableConfigurationProperties;
28  import org.springframework.boot.web.client.RestTemplateBuilder;
29  import org.springframework.context.annotation.Bean;
30  import org.springframework.context.annotation.Configuration;
31  import org.springframework.context.event.EventListener;
32  import org.springframework.security.oauth2.jwt.JwtDecoder;
33  import org.springframework.util.Assert;
34  import org.springframework.util.ClassUtils;
35  
36  /**
37   * The authentication support auto configuration.
38   *
39   * @author Christian Bremer
40   */
41  @ConditionalOnWebApplication(type = Type.SERVLET)
42  @ConditionalOnProperty(
43      prefix = "bremersee.security.authentication",
44      name = "enable-jwt-support",
45      havingValue = "true")
46  @ConditionalOnClass({
47      org.springframework.boot.web.client.RestTemplateBuilder.class,
48      org.bremersee.security.authentication.JsonPathJwtConverter.class,
49      org.bremersee.security.authentication.RestTemplateAccessTokenRetriever.class
50  })
51  @EnableConfigurationProperties(AuthenticationProperties.class)
52  @Configuration
53  @Slf4j
54  public class AuthenticationSupportAutoConfiguration {
55  
56    private AuthenticationProperties properties;
57  
58    /**
59     * Instantiates a new authentication support auto configuration.
60     *
61     * @param properties the properties
62     */
63    public AuthenticationSupportAutoConfiguration(
64        AuthenticationProperties properties) {
65      this.properties = properties;
66    }
67  
68    /**
69     * Init.
70     */
71    @EventListener(ApplicationReadyEvent.class)
72    public void init() {
73      log.info("\n"
74              + "*********************************************************************************\n"
75              + "* {}\n"
76              + "*********************************************************************************\n"
77              + "* rolesJsonPath = {}\n"
78              + "* rolesValueList = {}\n"
79              + "* rolesValueSeparator = {}\n"
80              + "* rolePrefix = {}\n"
81              + "* nameJsonPath = {}\n"
82              + "*********************************************************************************",
83          ClassUtils.getUserClass(getClass()).getSimpleName(),
84          properties.getRolesJsonPath(),
85          properties.isRolesValueList(),
86          properties.getRolesValueSeparator(),
87          properties.getRolePrefix(),
88          properties.getNameJsonPath());
89    }
90  
91    /**
92     * Json path jwt converter json path jwt converter.
93     *
94     * @return the json path jwt converter
95     */
96    @ConditionalOnMissingBean
97    @Bean
98    public JsonPathJwtConverter jsonPathJwtConverter() {
99      log.info("Creating {} ...", JsonPathJwtConverter.class.getSimpleName());
100     JsonPathJwtConverter converter = new JsonPathJwtConverter();
101     converter.setNameJsonPath(properties.getNameJsonPath());
102     converter.setRolePrefix(properties.getRolePrefix());
103     converter.setRolesJsonPath(properties.getRolesJsonPath());
104     converter.setRolesValueList(properties.isRolesValueList());
105     converter.setRolesValueSeparator(properties.getRolesValueSeparator());
106     return converter;
107   }
108 
109   /**
110    * Rest template access token retriever rest template access token retriever.
111    *
112    * @param restTemplateBuilder the rest template builder
113    * @return the rest template access token retriever
114    */
115   @ConditionalOnMissingBean
116   @Bean
117   public RestTemplateAccessTokenRetriever restTemplateAccessTokenRetriever(
118       ObjectProvider<RestTemplateBuilder> restTemplateBuilder) {
119 
120     Assert.notNull(
121         restTemplateBuilder.getIfAvailable(),
122         "Rest template builder must be present.");
123     log.info("Creating {} ...", RestTemplateAccessTokenRetriever.class.getSimpleName());
124     return new RestTemplateAccessTokenRetriever(restTemplateBuilder.getIfAvailable().build());
125   }
126 
127   /**
128    * Password flow authentication manager password flow authentication manager.
129    *
130    * @param jwtDecoder the jwt decoder
131    * @param jwtConverter the jwt converter
132    * @param tokenRetriever the token retriever
133    * @return the password flow authentication manager
134    */
135   @ConditionalOnMissingBean
136   @Bean
137   public PasswordFlowAuthenticationManager passwordFlowAuthenticationManager(
138       ObjectProvider<JwtDecoder> jwtDecoder,
139       JsonPathJwtConverter jwtConverter,
140       RestTemplateAccessTokenRetriever tokenRetriever) {
141 
142     Assert.notNull(
143         jwtDecoder.getIfAvailable(),
144         "Jwt decoder must be present.");
145 
146     log.info("Creating {} ...", PasswordFlowAuthenticationManager.class.getSimpleName());
147     return new PasswordFlowAuthenticationManager(
148         properties,
149         jwtDecoder.getIfAvailable(),
150         jwtConverter,
151         tokenRetriever);
152   }
153 
154 }
