17  package org.bremersee.context;
18  
19  import java.nio.charset.StandardCharsets;
20  import java.util.ArrayList;
21  import java.util.HashMap;
22  import java.util.List;
23  import java.util.Map;
24  import lombok.EqualsAndHashCode;
25  import lombok.Getter;
26  import lombok.Setter;
27  import lombok.ToString;
28  import org.springframework.boot.context.properties.ConfigurationProperties;
29  
30  /**
31   * The message source properties.
32   *
33   * @author Christian Bremer
34   */
35  @ConfigurationProperties("bremersee.messages")
36  @Getter
37  @Setter
38  @ToString
39  @EqualsAndHashCode
40  public class MessageSourceProperties {
41  
42    /**
43     * Set whether to always apply the {@code MessageFormat} rules, parsing even messages without
44     * arguments.
45     *
46     * <p>Default is "false": Messages without arguments are by default
47     * returned as-is, without parsing them through MessageFormat. Set this to "true" to enforce
48     * MessageFormat for all messages, expecting all message texts to be written with MessageFormat
49     * escaping.
50     *
51     * <p>For example, MessageFormat expects a single quote to be escaped
52     * as "''". If your message texts are all written with such escaping, even when not defining
53     * argument placeholders, you need to set this flag to "true". Else, only message texts with
54     * actual arguments are supposed to be written with MessageFormat escaping.
55     *
56     * @see java.text.MessageFormat
57     */
58    private boolean alwaysUseMessageFormat = false; // from MessageSourceSupport
59  
60    /**
61     * Set whether to use the message code as default message instead of throwing a
62     * NoSuchMessageException. Useful for development and debugging. Default is "false".
63     */
64    private boolean useCodeAsDefaultMessage = false; // from AbstractMessageSource
65  
66    /**
67     * Set an list of basenames, each following the basic ResourceBundle convention of not specifying
68     * file extension or language codes. The resource location format is up to the specific {@code
69     * MessageSource} implementation.
70     *
71     * <p>Regular and XMl properties files are supported: e.g. "messages" will find
72     * a "messages.properties", "messages_en.properties" etc arrangement as well as "messages.xml",
73     * "messages_en.xml" etc.
74     *
75     * <p>The associated resource bundles will be checked sequentially when resolving
76     * a message code. Note that message definitions in a <i>previous</i> resource bundle will
77     * override ones in a later bundle, due to the sequential lookup.
78     */
79    private List<String> baseNames = new ArrayList<>(); // from AbstractResourceBasedMessageSource
80  
81    /**
82     * Set the number of seconds to cache loaded properties files.
83     * <ul>
84     * <li>Default is "-1", indicating to cache forever (just like
85     * {@code java.util.ResourceBundle}).
86     * <li>A positive number will cache loaded properties files for the given
87     * number of seconds. This is essentially the interval between refresh checks.
88     * Note that a refresh attempt will first check the last-modified timestamp
89     * of the file before actually reloading it; so if files don't change, this
90     * interval can be set rather low, as refresh attempts will not actually reload.
91     * <li>A value of "0" will check the last-modified timestamp of the file on
92     * every message access. <b>Do not use this in a production environment!</b>
93     * </ul>
94     */
95    private int cacheSeconds = -1; // from AbstractResourceBasedMessageSource
96  
97    /**
98     * Set the default charset to use for parsing properties files. Used if no file-specific charset
99     * is specified for a file.
100    *
101    * <p>The effective default is the {@code java.util.Properties}
102    * default encoding: ISO-8859-1. A {@code null} value indicates the platform default encoding.
103    *
104    * <p>Only applies to classic properties files, not to XML files.
105    */
106   private String defaultEncoding = StandardCharsets.UTF_8.name(); // from AbstractResourceBasedM.
107 
108   /**
109    * Set whether to fall back to the system Locale if no files for a specific Locale have been
110    * found. Default is "true"; if this is turned off, the only fallback will be the default file
111    * (e.g. "messages.properties" for basename "messages").
112    *
113    * <p>Falling back to the system Locale is the default behavior of
114    * {@code java.util.ResourceBundle}. However, this is often not desirable in an application server
115    * environment, where the system Locale is not relevant to the application at all: set this flag
116    * to "false" in such a scenario.
117    */
118   private boolean fallbackToSystemLocale = true;
119 
120   /**
121    * Specify a default Locale to fall back to, as an alternative to falling back to the system
122    * Locale.
123    *
124    * <p>Default is to fall back to the system Locale. You may override this with
125    * a locally specified default Locale here, or enforce no fallback locale at all through
126    * disabling.
127    */
128   private String defaultLocale;
129 
130   /**
131    * Specifies whether the resource bundles are reloadable or not.
132    */
133   private boolean useReloadableMessageSource = false;
134 
135   /**
136    * Specify whether to allow for concurrent refresh behavior, i.e. one thread locked in a refresh
137    * attempt for a specific cached properties file whereas other threads keep returning the old
138    * properties for the time being, until the refresh attempt has completed.
139    *
140    * <p>Default is "true": this behavior is new as of Spring Framework 4.1,
141    * minimizing contention between threads. If you prefer the old behavior, i.e. to fully block on
142    * refresh, switch this flag to "false".
143    */
144   private boolean concurrentRefresh = true; // ReloadableResourceBundleMessageSource
145 
146   /**
147    * Set per-file charsets to use for parsing properties files.
148    *
149    * <p>Only applies to classic properties files, not to XML files.
150    */
151   private Map<String, String> fileEncodings = new HashMap<>(); // ReloadableResourceBundleMessageS.
152 
153 }
