34  /**
35   * The auto configure test.
36   *
37   * @author Christian Bremer
38   */
39  @SpringBootTest(
40      classes = TestConfiguration.class,
41      webEnvironment = WebEnvironment.NONE,
42      properties = {
43          "bremersee.messages.fallback-to-system-locale=false",
44          "bremersee.messages.default-locale=en",
45          "bremersee.messages.base-names=i18n"
46      })
47  @Slf4j
48  public class AutoConfigureTest {
49  
50    /**
51     * The message source.
52     */
53    @Autowired
54    MessageSource messageSource;
55  
56    @Autowired(required = false)
57    AclMapper<AclEntity> aclMapper;
58  
59    @Autowired
60    ModelMapper modelMapper;
61  
62    /**
63     * Test message source.
64     */
65    @Test
66    void testMessageSource() {
67      String actual = messageSource.getMessage("test0", null, Locale.ENGLISH);
68      assertEquals("Hello JUnit", actual);
69  
70      actual = messageSource.getMessage("test1", new Object[]{"Anna"}, Locale.ENGLISH);
71      assertEquals("Hello Anna", actual);
72  
73      actual = messageSource.getMessage("test0", null, Locale.GERMAN);
74      assertEquals("Hallo JUnit", actual);
75  
76      actual = messageSource.getMessage("test1", new Object[]{"Anna"}, Locale.GERMAN);
77      assertEquals("Hallo Anna", actual);
78  
79      actual = messageSource.getMessage("test0", null, Locale.FRANCE);
80      assertEquals("Bon jour JUnit", actual);
81  
82      actual = messageSource.getMessage("test1", new Object[]{"Anna"}, Locale.FRANCE);
83      assertEquals("Bon jour Anna", actual);
84    }
85  
86    @Test
87    void testAclMapper() {
88      assertNotNull(modelMapper);
89    }

91  }
