18  /**
19   * The keycloak jwt converter test.
20   *
21   * @author Christian Bremer
22   */
23  public class KeycloakJwtConverterTest {
24  
25    /**
26     * Convert jwt.
27     */
28    @Test
29    public void convert() {
30      Map<String, Object> headers = new HashMap<>();
31      headers.put("alg", "RS256");
32      headers.put("kid", "-V5Nzr9xllRiyURYK-t6pB7C-5E8IKm8-eAPFLVvCt4");
33      headers.put("typ", "JWT");
34  
35      Date exp = new Date(System.currentTimeMillis() + 1000L * 60L * 15L);
36      Date iat = new Date();
37      String sub = UUID.randomUUID().toString();
38      JWTClaimsSet.Builder builder = new Builder();
39      builder.audience("account");
40      builder.expirationTime(exp);
41      builder.issuer("https://openid.dev.bremersee.org/auth/realms/omnia");
42      builder.issueTime(iat);
43      builder.jwtID("48502385-7f37-47da-abc9-a223b8972795");
44      builder.notBeforeTime(iat);
45      builder.subject(sub);
46  
47      builder.claim("scopes", "email profile");
48      builder.claim("email_verified", false);
49      builder.claim("name", "Anna Livia Plurabelle");
50      builder.claim("preferred_username", "anna");
51      builder.claim("given_name", "Anna Livia");
52      builder.claim("family_name", "Plurabelle");
53      builder.claim("email", "anna@example.org");
54  
55      Map<String, Object> roles = new LinkedHashMap<>();
56      roles.put("roles", Collections.singletonList("LOCAL_USER"));
57      builder.claim("realm_access", roles);
58  
59      JWTClaimsSet claimsSet = builder.build();
60      Jwt jwt = new Jwt(
61          "an-access-token",
62          iat.toInstant(),
63          exp.toInstant(),
64          headers,
65          claimsSet.getClaims());
66      KeycloakJwtConverter converter = new KeycloakJwtConverter();
67      KeycloakJwtAuthenticationToken auth = (KeycloakJwtAuthenticationToken) converter.convert(jwt);
68      assertNotNull(auth);
69      assertEquals("anna", auth.getPreferredName());
70      assertTrue(auth.getAuthorities().stream()
71          .anyMatch(grantedAuthority -> grantedAuthority.getAuthority()
72              .equalsIgnoreCase("ROLE_LOCAL_USER")));
73    }
74  }
