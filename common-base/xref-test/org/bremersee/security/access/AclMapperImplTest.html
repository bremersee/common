24    /**
25     * Map with defaults and admin switch.
26     */
27    @Test
28    public void mapWithDefaultsAndAdminSwitch() {
29      AclMapper<Acl<? extends Ace>> mapper = new AclMapperImpl<>(
30          AclImpl::new,
31          PermissionConstants.ALL,
32          true,
33          false
34      );
35  
36      assertNotNull(mapper.map((Acl<? extends Ace>) null));
37      assertNotNull(mapper.map((AccessControlList) null));
38  
39      AccessControlList source = AccessControlList
40          .builder()
41          .owner("owner")
42          .entries(Arrays.asList(
43              AccessControlEntry
44                  .builder()
45                  .permission("write")
46                  .groups(Collections.singletonList("group"))
47                  .roles(Collections.singletonList("role"))
48                  .users(Collections.singletonList("user"))
49                  .build(),
50              AccessControlEntry
51                  .builder()
52                  .permission("read")
53                  .groups(new ArrayList<>())
54                  .roles(new ArrayList<>())
55                  .users(new ArrayList<>())
56                  .guest(true)
57                  .build()
58          ))
59          .build();
60  
61      Acl<? extends Ace> destination = mapper.map(source);
62      assertNotNull(destination);
63      assertTrue(destination.entryMap().keySet().containsAll(Arrays.asList(PermissionConstants.ALL)));
64      assertEquals(source.getOwner(), destination.getOwner());
65      assertTrue(destination.entryMap().get("write").getUsers().contains("user"));
66      assertTrue(destination
67          .entryMap()
68          .get("write")
69          .getRoles()
70          .containsAll(Arrays.asList("role", AuthorityConstants.ADMIN_ROLE_NAME)));
71      assertTrue(destination.entryMap().get("write").getGroups().contains("group"));
72  
73      source = mapper.map(destination);
74      assertNotNull(source);
75      assertEquals(destination.getOwner(), source.getOwner());
76      for (String permission : PermissionConstants.ALL) {
77        assertTrue(source
78            .getEntries()
79            .stream()
80            .anyMatch(ace -> ace.getPermission().equals(permission)));
81      }
82      AccessControlEntry ace = source
83          .getEntries()
84          .stream()
85          .filter(entry -> entry.getPermission().equals("write"))
86          .findFirst()
87          .orElse(null);
88      assertNotNull(ace);
89      assertTrue(ace.getGroups().contains("group"));
90      assertTrue(ace.getRoles().contains("role"));
91      assertTrue(ace.getUsers().contains("user"));
92      assertFalse(ace.getRoles().contains(AuthorityConstants.ADMIN_ROLE_NAME));
93  
94      AccessControlList accessControlList = mapper.defaultAccessControlList("owner");
95      assertNotNull(accessControlList);
96      assertEquals("owner", accessControlList.getOwner());
97  
98      Acl<? extends Ace> acl = mapper.defaultAcl("owner");
99      assertNotNull(acl);
100     assertEquals("owner", acl.getOwner());
101   }

106   @Test
107   public void map() {
108     AclMapper<Acl<? extends Ace>> mapper = new AclMapperImpl<>(
109         AclImpl::new,
110         null,
111         false,
112         true
113     );
114 
115     assertNull(mapper.map((Acl<? extends Ace>) null));
116     assertNull(mapper.map((AccessControlList) null));
117 
118     AccessControlList source = AccessControlList
119         .builder()
120         .owner("owner")
121         .entries(Arrays.asList(
122             AccessControlEntry
123                 .builder()
124                 .permission("write")
125                 .groups(Collections.singletonList("group"))
126                 .roles(Collections.singletonList("role"))
127                 .users(Collections.singletonList("user"))
128                 .build(),
129             AccessControlEntry
130                 .builder()
131                 .permission("read")
132                 .groups(new ArrayList<>())
133                 .roles(new ArrayList<>())
134                 .users(new ArrayList<>())
135                 .guest(true)
136                 .build()
137         ))
138         .build();
139 
140     Acl<? extends Ace> destination = mapper.map(source);
141     assertNotNull(destination);
142     assertFalse(
143         destination.entryMap().keySet().containsAll(Arrays.asList(PermissionConstants.ALL)));
144     assertEquals(source.getOwner(), destination.getOwner());
145     assertTrue(destination.entryMap().get("write").getUsers().contains("user"));
146     assertFalse(destination
147         .entryMap()
148         .get("write")
149         .getRoles()
150         .containsAll(Arrays.asList("role", AuthorityConstants.ADMIN_ROLE_NAME)));
151     assertTrue(destination.entryMap().get("write").getGroups().contains("group"));
152 
153     source = mapper.map(destination);
154     assertNotNull(source);
155     assertEquals(destination.getOwner(), source.getOwner());
156     assertFalse(source
157         .getEntries()
158         .stream()
159         .anyMatch(ace -> ace.getPermission().equals(PermissionConstants.DELETE)));
160     AccessControlEntry ace = source
161         .getEntries()
162         .stream()
163         .filter(entry -> entry.getPermission().equals("write"))
164         .findFirst()
165         .orElse(null);
166     assertNotNull(ace);
167     assertTrue(ace.getGroups().contains("group"));
168     assertTrue(ace.getRoles().contains("role"));
169     assertTrue(ace.getUsers().contains("user"));
170     assertFalse(ace.getRoles().contains(AuthorityConstants.ADMIN_ROLE_NAME));
171   }
