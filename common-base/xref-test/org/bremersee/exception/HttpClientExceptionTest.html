19    /**
20     * Tests creation.
21     */
22    @Test
23    public void create() {
24      MultiValueMap<String, String> headers = new LinkedMultiValueMap<>();
25      headers.add("foo", "bar");
26      RestApiException restApiException = new RestApiException();
27      HttpClientException exception = new HttpClientException(
28          404,
29          "Not found",
30          headers,
31          restApiException);
32      assertEquals(404, exception.status());
33      assertEquals("Not found", exception.getMessage());
34      assertTrue(exception.getMultiValueHeaders().containsKey("foo"));
35      assertEquals(restApiException, exception.getRestApiException());
36      assertNotNull(exception.toString());
37      assertEquals(exception, new HttpClientException(
38          404,
39          "Not found",
40          headers,
41          restApiException));
42    }
