17  package org.bremersee.security.authentication;
18  
19  import com.jayway.jsonpath.DocumentContext;
20  import com.jayway.jsonpath.JsonPath;
21  import com.jayway.jsonpath.Option;
22  import java.util.Collections;
23  import java.util.List;
24  import java.util.Set;
25  import java.util.stream.Collectors;
26  import net.minidev.json.JSONValue;
27  import org.springframework.core.convert.converter.Converter;
28  import org.springframework.security.core.GrantedAuthority;
29  import org.springframework.security.core.authority.SimpleGrantedAuthority;
30  import org.springframework.security.oauth2.jwt.Jwt;
31  import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationToken;
32  import org.springframework.util.StringUtils;
33  
34  /**
35   * A JWT converter for the keycloak identity provider. It use the json path {@code
36   * $.realm_access.roles} to extract the roles from the token.
37   *
38   * @author Christian Bremer
39   */
40  public class KeycloakJwtConverter implements Converter<Jwt, JwtAuthenticationToken> {
41  
42    private static final String JSON_ROLES_PATH = "$.realm_access.roles";
43  
44    private static final com.jayway.jsonpath.Configuration jsonPathConf
45        = com.jayway.jsonpath.Configuration.builder().options(Option.SUPPRESS_EXCEPTIONS).build();
46  
47    /**
48     * Instantiates a new keycloak jwt converter.
49     */
50    @SuppressWarnings("WeakerAccess")
51    public KeycloakJwtConverter() {
52    }
53  
54    @Override
55    public JwtAuthenticationToken convert(final Jwt jwt) {
56      final String jsonStr = JSONValue.toJSONString(jwt.getClaims());
57      final DocumentContext ctx = JsonPath.parse(jsonStr, jsonPathConf);
58      final List<?> roles = ctx.read(JSON_ROLES_PATH, List.class);
59      final Set<GrantedAuthority> authorities;
60      if (roles == null || roles.isEmpty()) {
61        authorities = Collections.emptySet();
62      } else {
63        authorities = roles
64            .stream()
65            .map(this::buildGrantedAuthority)
66            .collect(Collectors.toSet());
67      }
68      return new KeycloakJwtAuthenticationToken(jwt, authorities);
69    }
70  
71    private GrantedAuthority buildGrantedAuthority(final Object name) {
72      final SimpleGrantedAuthority authority;
73      if (name instanceof String && StringUtils.hasText((String) name)) {
74        authority = new SimpleGrantedAuthority(((String) name).startsWith("ROLE_")
75            ? (String) name : "ROLE_" + name);
76      } else {
77        authority = new SimpleGrantedAuthority("ROLE_NULL");
78      }
79      return authority;
80    }
81  }
